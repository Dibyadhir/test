--- create database wexa
--- create table user(id int not null primary key auto_increment, password varchar(300) not null, email varchar(300) not null, pic varchar(300) not null, first_name varchar(300) not null, last_name varchar(300) not null, gender varchar(20) not null, dob varchar(20) not null, about_yourself varchar(500) not null);
--- CREATE TABLE activity_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    activity_type ENUM('login','profile_update', 'friend_request', 'friend_accepted') NOT NULL,
    activity_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    details TEXT,
    FOREIGN KEY (user_id) REFERENCES user(id)
);
